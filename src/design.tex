\section{\sysname Design}
\label{sec:design}

\subsection{Overview}
\label{sec:design:overview}

\sysname resolves the useful-compute ratio vs.\ throughput tension
(Section~\ref{sec:bg:gap}) through a two-stage design
(Figure~\ref{fig:overview}):

\begin{enumerate}
  \item \textbf{Construction stage.} A lightweight kernel scans the input
        active mask $M_{in}$ and evaluates Eq.~\eqref{eq:mask} for each
        output position, emitting an ordered \emph{worklist}
        $\mathbf{q} = (q_1, \ldots, q_m)$ of $m = \|M\|_0$ active output
        indices drawn from $G_{out}$. The worklist preserves tile-local
        spatial locality (Section~\ref{sec:design:construct}).

  \item \textbf{Compute stage.} A convolution kernel consumes $\mathbf{q}$
        as its coordinate source. For each $q_j$, the kernel decodes the
        spatial coordinate, reads the receptive-field input
        $\mathbf{x}[\mathcal{N}(q_j)]$, and accumulates
        Eq.~\eqref{eq:conv} identically to the dense pipeline
        (Section~\ref{sec:design:compute}).
\end{enumerate}

The computed set $\mathcal{C} = \{q_1, \ldots, q_m\}$ contains only active
positions (high useful-compute ratio), while the per-position datapath
remains that of the dense pipeline (near-dense throughput). The
construction stage operates entirely in mask space and its cost is
independent of channel dimensions, making its overhead negligible relative
to the compute stage.

\subsection{Worklist Construction}
\label{sec:design:construct}

The construction stage produces the worklist $\mathbf{q}$ from the input
mask $M_{in}$ by performing a per-tile gather followed by concatenation.
It operates on the tile partition $T_1, \ldots, T_N$ defined in
Eq.~\eqref{eq:tile}. For each output position, the stage first evaluates
Eq.~\eqref{eq:mask} to determine the output mask $M$ (requiring $k^2$
boolean reads of $M_{in}$ per position), then gathers the active indices
tile by tile.

\paragraph{Per-tile gather.}
For each tile $T_n$, the output mask $M$ induces a support set of
active indices within the tile:
\begin{equation}
  \mathcal{I}_n = \bigl\{\, k \in \{1, \ldots, t_h t_w\} : M[T_n[k]] = 1 \,\bigr\}.
  \label{eq:support}
\end{equation}
The per-tile subsequence is the gather of $T_n$ by its support set:
\begin{equation}
  \mathbf{q}_n = (T_n)_{\mathcal{I}_n}
               = (T_n[k])_{k \in \mathcal{I}_n},
  \label{eq:pertile-gather}
\end{equation}
ordered by ascending $k$ (i.e., row-major within the tile), with length
$|\mathbf{q}_n| = |\mathcal{I}_n|$.

\paragraph{Worklist concatenation.}
Let $P = (P_1, \ldots, P_N)$ be the permutation of $\{1, \ldots, N\}$
determined by nondeterministic parallel block scheduling. The worklist is:
\begin{equation}
  \mathbf{q} = (\mathbf{q}_{P_1},\; \mathbf{q}_{P_2},\; \ldots,\;
               \mathbf{q}_{P_N}),
  \label{eq:worklist}
\end{equation}
with total length $|\mathbf{q}| = \|M\|_0$. Each $\mathbf{q}_{P_n}$
retains its internal row-major ordering; only the inter-tile order varies
with $P$.


\paragraph{Cost analysis.}
Mask evaluation (Eq.~\ref{eq:mask}) performs $k^2$ boolean reads per
output position, totaling $O(H_{out} W_{out} k^2)$ work. The gather and
concatenation write $\|M\|_0$ integer indices. The entire stage operates
in 2D mask space: its cost depends on spatial resolution and kernel size
but is independent of $C_{in}$ and $C_{out}$. Because the compute stage
scales with $\|M\|_0 \cdot k^2 \cdot C_{in} \cdot C_{out}$, the
construction overhead becomes negligible as channel dimensions grow
(measured at XX\% in Section~\ref{sec:eval}).

\begin{algorithm}[t]
\caption{Worklist Construction}
\label{alg:construct}
\begin{algorithmic}[1]
\REQUIRE Tiles $T_1, \ldots, T_N$, input mask $M_{in}$
\ENSURE Worklist $\mathbf{q}$, output mask $M$
\FOR{each tile $T_n$ \textbf{in parallel}}
  \FOR{each $k \in \{1, \ldots, t_h t_w\}$}
    \STATE $g \leftarrow T_n[k]$
    \STATE $M[g] \leftarrow$ Eq.~\eqref{eq:mask}
  \ENDFOR
  \STATE $\mathcal{I}_n \leftarrow \{k : M[T_n[k]] = 1\}$
  \STATE $\mathbf{q}_n \leftarrow (T_n[k])_{k \in \mathcal{I}_n}$
\ENDFOR
\STATE $\mathbf{q} \leftarrow (\mathbf{q}_{P_1}, \ldots, \mathbf{q}_{P_N})$
  \hfill \COMMENT{$P$ from block scheduling}
\end{algorithmic}
\end{algorithm}


\subsection{Worklist-driven Convolution}
\label{sec:design:compute}

The compute stage evaluates Eq.~\eqref{eq:conv} for exactly the positions
in $\mathbf{q}$, formulated as an implicit GEMM whose M-dimension is
indexed by the worklist rather than a full raster scan of $G_{out}$.

\paragraph{GEMM formulation.}
Let $m = |\mathbf{q}| = \|M\|_0$. The masked convolution computes:
\begin{equation}
  \mathbf{Y}_{m \times C_{out}} = \mathbf{A}_{m \times k^2 C_{in}}
  \;\cdot\; \hat{\mathbf{W}}_{k^2 C_{in} \times C_{out}},
  \label{eq:gemm}
\end{equation}
where $\hat{\mathbf{W}}$ is the weight tensor reshaped into a 2D matrix,
and row $j$ of $\mathbf{A}$ is the flattened receptive-field input for
worklist entry $q_j$:
\begin{equation}
  \mathbf{A}[j, :] = \bigl[\,
    \mathbf{x}[\mathcal{N}_1(q_j)]^\top,\;
    \ldots,\;
    \mathbf{x}[\mathcal{N}_{k^2}(q_j)]^\top
  \,\bigr].
  \label{eq:implicit-a}
\end{equation}
The matrix $\mathbf{A}$ is never materialized; each element is computed on
the fly by decoding the spatial coordinate from $q_j$ and reading the
corresponding input position from $\mathbf{x}$. The execution follows the
standard implicit-GEMM tiled pipeline; the only modification is the
coordinate source: the worklist $\mathbf{q}$ replaces the sequential scan
of $G_{out}$. After accumulation, each output row is written back to
$\mathbf{y}[q_j]$ using the worklist entry as the output address.
Algorithm~\ref{alg:compute} summarizes this procedure.

\paragraph{Locality.}
Each compute tile processes $B_M$ consecutive worklist entries. Because
construction compacts active positions per tile, consecutive entries in
each $\mathbf{q}_{P_n}$ are row-major neighbors in $G_{out}$ whose
receptive fields overlap in contiguous memory. A compute chunk of $B_M$
entries spans at most $\lceil B_M / \min_n |\mathbf{q}_{P_n}| \rceil$
construction tiles regardless of the permutation $P$; the
nondeterministic inter-tile ordering does not degrade spatial locality
relative to the dense raster order.

\paragraph{Equivalence to dense execution.}
For any single position $q_j$, the operations (decode coordinate, read
$k^2 \times C_{in}$ input values, multiply-accumulate against
$\mathbf{W}$, write $C_{out}$ outputs) are identical to what the dense
pipeline performs for that position. \sysname changes only \emph{which}
positions are processed ($\mathbf{q}$ instead of the full $G_{out}$) and
\emph{how many} ($\|M\|_0$ instead of $H_{out} W_{out}$). The arithmetic
intensity, memory access pattern per position, and GPU occupancy model are
preserved.

\begin{algorithm}[t]
\caption{Worklist-driven Convolution}
\label{alg:compute}
\begin{algorithmic}[1]
\REQUIRE Worklist $\mathbf{q}$, input $\mathbf{x}$, weight $\hat{\mathbf{W}}$
\ENSURE Output $\mathbf{y}$ at positions in $\mathbf{q}$
\FOR{each tile $(b_m, b_n)$ \textbf{in parallel}}
  \STATE Load worklist segment $\mathbf{q}[b_m]$
  \STATE Initialize accumulator $\mathbf{Y} \leftarrow \mathbf{0}_{B_M \times B_N}$
  \FOR{$k = 1, \ldots, \lceil k^2 C_{in} / B_K \rceil$}
    \STATE $\mathbf{X} \leftarrow$ gather input block from $\mathbf{x}$ at positions $\mathcal{N}(\mathbf{q}[b_m])$, channel tile $k$
    \STATE $\hat{\mathbf{W}}_k \leftarrow$ load weight block for channel tile $k$
    \STATE $\mathbf{Y} \leftarrow \mathbf{Y} + \mathbf{X} \cdot \hat{\mathbf{W}}_k$
  \ENDFOR
  \STATE Write $\mathbf{Y}$ to $\mathbf{y}$ at addresses $\mathbf{q}[b_m]$
\ENDFOR
\end{algorithmic}
\end{algorithm}
